namespace EstudoCSharp;

class Program
{
    static void Main(string[] args)
    {
        Apresentar();

        decimal salario = PedirValor("Informe o seu salário base:");
        decimal bonus = PedirValor("Informe o bonus em porcentagem:");
        decimal desconto = PedirValor("Informe o desconto em reais:");

        decimal conta = FazerConta(salario, bonus, desconto);

        FormularResposta(conta);
    }

    public static void Apresentar()
    {
        Console.WriteLine("====================================");
        Console.WriteLine("==  Programa do Salário Líquido   ==");
        Console.WriteLine("====================================");
    }

    public static int PedirValor(string msg)
    {
        Console.WriteLine(msg);
        int valor = Convert.ToInt32(Console.ReadLine());
        return valor;
    }

    public static decimal FazerConta(decimal salario, decimal bonus, decimal desconto)
    {
        decimal bonusc = bonus / 100;
        decimal conta = (salario + (salario * bonusc)) - desconto;
        return conta;
    }

    public static void FormularResposta(decimal conta)
    {
        Console.WriteLine($"Seu salário líquído é de R$ {conta.ToString("0.00")}");
    }
}
