namespace EstudoCSharp;

class Program
{
    static void Main(string[] args)
    {
        Apresentar();

        decimal tanque = PedirValor("Informe a quantidade máxima do tanque: ");
        decimal consumo = PedirValor("Informe o consumo do veículo por km rodado:");
        decimal distancia = PedirValor("Informe a distância da Viagem:");

        int conta = FazerConta(tanque, consumo, distancia);

        FormularResposta(conta);
    }

    public static void Apresentar()
    {
        Console.WriteLine("===========================");
        Console.WriteLine("==  Programa da Viagem   ==");
        Console.WriteLine("===========================");
    }

    public static decimal PedirValor(string msg)
    {
        Console.WriteLine(msg);
        decimal valor = Convert.ToDecimal(Console.ReadLine());
        return valor;
    }

    public static int FazerConta(decimal tanque, decimal consumo, decimal distancia)
    {
        decimal result = distancia / (tanque * consumo);
        int conta = (int)result + 1;
        return conta;
    }

    public static void FormularResposta(int conta)
    {
        Console.WriteLine($"Você precisará fazer {conta} paradas para abastecer.");
    }
}
